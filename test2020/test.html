    cleaned = ''.join(ch if ch.isalnum() or ch.isspace() else ' ' for ch in sentence)
    words = cleaned.split()
    return max(words, key=len) if words else ''

def convert_days(total_days):
    if total_days < 0:
        return "Invalid Input"
    years = total_days // 365
    remaining = total_days % 365
    months = remaining // 30
    days = remaining % 30
    return {"years": years, "months": months, "days": days}

print(convert_days(1122))
print(convert_days(-5))
print(convert_days(400))

def compare_lists(list1, list2):
    s1, s2 = set(list1), set(list2)
    return {
        "common": s1 & s2,
        "only_in_first": s1 - s2,
        "only_in_second": s2 - s1

def fibonacci(n):
    if n <= 0: return []
    seq = [0, 1]
    for _ in range(2, n):
        seq.append(seq[-1] + seq[-2])
    return seq[:n]

    words = text.lower().split()
    freq = {}
    for w in words:
        freq[w] = freq.get(w, 0) + 1

    if not lst: return lst
    k = k % len(lst)
    return lst[-k:] + lst[:-k]

    result = {}
    for w in words:
        key = len(w)
        result.setdefault(key, []).append(w)
    return result
